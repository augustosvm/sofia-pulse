```sql
SELECT
  research_area,
  COUNT(DISTINCT grant_id) as num_grants,
  COUNT(DISTINCT paper_id) as num_papers,
  (COUNT(DISTINCT grant_id) / NULLIF(COUNT(DISTINCT paper_id), 0)) as grant_paper_ratio
FROM (
  SELECT g.research_area, g.project_number as grant_id, NULL as paper_id
  FROM nih_grants g
  WHERE g.fiscal_year >= 2023
  UNION ALL
  SELECT
    CASE
      WHEN p.keywords @> ARRAY['CRISPR'] THEN 'Gene Editing'
      WHEN p.keywords @> ARRAY['mRNA'] THEN 'mRNA Therapeutics'
      -- ... more mappings
    END as research_area,
    NULL as grant_id,
    p.id as paper_id
  FROM openalex_papers p
  WHERE p.publication_year >= 2023
) combined
GROUP BY research_area
HAVING COUNT(DISTINCT grant_id) > 50 AND COUNT(DISTINCT paper_id) < 100
ORDER BY grant_paper_ratio DESC;
```

```sql
WITH research_momentum AS (
  SELECT
    keywords,
    COUNT(*) FILTER (WHERE published_date >= NOW() - INTERVAL '6 months') as papers_6mo,
    COUNT(*) FILTER (WHERE published_date >= NOW() - INTERVAL '12 months') as papers_12mo
  FROM arxiv_ai_papers
  WHERE is_breakthrough = true
  GROUP BY keywords
),
grant_funding AS (
  SELECT
    research_area,
    COUNT(*) as num_grants,
    SUM(award_amount_usd) as total_funding
  FROM nih_grants
  WHERE fiscal_year >= 2023
  GROUP BY research_area
)
SELECT
  rm.keywords as research_topic,
  rm.papers_6mo,
  rm.papers_12mo,
  (rm.papers_6mo::float / rm.papers_12mo) as growth_rate,
  gf.num_grants,
  gf.total_funding,
  (gf.num_grants::float / rm.papers_6mo) as grant_per_paper_ratio
FROM research_momentum rm
JOIN grant_funding gf ON gf.research_area = ANY(rm.keywords)
WHERE rm.papers_6mo > 20
  AND (gf.num_grants::float / rm.papers_6mo) > 2.0  -- Mais grants que papers = oportunidade
ORDER BY grant_per_paper_ratio DESC, growth_rate DESC
LIMIT 20;
```

```sql
WITH paper_momentum AS (
  SELECT
    unnest(keywords) as keyword,
    COUNT(*) as paper_count,
    COUNT(*) FILTER (WHERE published_date >= NOW() - INTERVAL '12 months') as papers_last_year
  FROM arxiv_ai_papers

SELECT
  pm.keyword,
  pm.paper_count,
  pm.papers_last_year,
  COALESCE(sc.num_companies, 0) as num_startups,
```

```sql
WITH skill_demand AS (
```
